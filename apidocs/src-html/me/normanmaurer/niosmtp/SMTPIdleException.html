001    package me.normanmaurer.niosmtp;
002    
003    /**
004     * {@link SMTPException} which will get thrown if the connection did idle for to long.
005     * This most times means that the SMTP Server did not send a {@link SMTPResponse} in a timely manner
006     * 
007     * @author Norman Maurer
008     *
009     */
010    public class SMTPIdleException extends SMTPException{
011    
012        /**
013         * 
014         */
015        private static final long serialVersionUID = 2383943619694157245L;
016    
017        public SMTPIdleException() {
018            super();
019        }
020    
021        public SMTPIdleException(String message, Throwable cause) {
022            super(message, cause);
023        }
024    
025        public SMTPIdleException(String message) {
026            super(message);
027        }
028    
029        public SMTPIdleException(Throwable cause) {
030            super(cause);
031        }
032    
033    }
