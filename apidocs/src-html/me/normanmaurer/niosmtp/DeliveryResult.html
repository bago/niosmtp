001    package me.normanmaurer.niosmtp;
002    
003    import java.util.Iterator;
004    
005    /**
006     * Result of an email delivery
007     * 
008     * @author Norman Maurer
009     *
010     */
011    public interface DeliveryResult {
012    
013        /**
014         * Return true if the delivery was successful (no exception)
015         * 
016         * @return success
017         */
018        public boolean isSuccess();
019        
020        /**
021         * Return the {@link SMTPException} which was thrown while try to deliver
022         * 
023         * @return exception
024         */
025        public SMTPException getException();
026        
027        /**
028         * Return an {@link Iterator} which holds all {@link DeliveryRecipientStatus} objects for the 
029         * delivery. This MAY return null if {@link #getCause()} returns not null
030         * 
031         * @return status
032         */
033        public Iterator<DeliveryRecipientStatus> getRecipientStatus();
034    }
