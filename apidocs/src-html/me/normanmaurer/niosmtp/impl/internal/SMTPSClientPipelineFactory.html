001    package me.normanmaurer.niosmtp.impl.internal;
002    
003    import javax.net.ssl.SSLContext;
004    import javax.net.ssl.SSLEngine;
005    
006    import org.jboss.netty.channel.ChannelPipeline;
007    import org.jboss.netty.channel.ChannelPipelineFactory;
008    import org.jboss.netty.handler.ssl.SslHandler;
009    
010    /**
011     * {@link ChannelPipelineFactory} which is used for SMTPS connections
012     * 
013     * @author Norman Maurer
014     * 
015     *
016     */
017    public class SMTPSClientPipelineFactory extends SMTPClientPipelineFactory{
018    
019        private final SSLContext context;
020    
021        public SMTPSClientPipelineFactory(SSLContext context) {
022            this.context = context;
023        }
024        
025        @Override
026        public ChannelPipeline getPipeline() throws Exception {
027            ChannelPipeline cp = super.getPipeline();
028            SSLEngine engine = context.createSSLEngine();
029            engine.setUseClientMode(true);
030            SslHandler sslHandler = new SslHandler(engine);
031            cp.addFirst("sslHandler", sslHandler);
032            return cp;
033        }
034    
035    }
