001    package me.normanmaurer.niosmtp.impl.internal;
002    
003    import java.util.Iterator;
004    
005    import me.normanmaurer.niosmtp.DeliveryRecipientStatus;
006    import me.normanmaurer.niosmtp.DeliveryResult;
007    import me.normanmaurer.niosmtp.SMTPException;
008    
009    public class DeliveryResultImpl implements DeliveryResult{
010    
011        private final Iterable<DeliveryRecipientStatus> status;
012        private final SMTPException exception;
013    
014        public DeliveryResultImpl(Iterable<DeliveryRecipientStatus> status) {
015            this.status = status;
016            this.exception = null;
017        }
018        
019        public DeliveryResultImpl(SMTPException exception) {
020            this.exception = exception;
021            this.status = null;
022        }
023        
024        @Override
025        public boolean isSuccess() {
026            return exception == null;
027        }
028    
029        @Override
030        public SMTPException getException() {
031            return exception;
032        }
033    
034        @Override
035        public Iterator<DeliveryRecipientStatus> getRecipientStatus() {
036            if (status == null) {
037                return null;
038            }
039            return status.iterator();
040        }
041    
042    }
