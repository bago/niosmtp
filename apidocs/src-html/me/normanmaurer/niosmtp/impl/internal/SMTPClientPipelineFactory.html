001    package me.normanmaurer.niosmtp.impl.internal;
002    
003    import org.jboss.netty.channel.ChannelPipeline;
004    import org.jboss.netty.channel.ChannelPipelineFactory;
005    import org.jboss.netty.channel.Channels;
006    import org.jboss.netty.handler.codec.frame.DelimiterBasedFrameDecoder;
007    import org.jboss.netty.handler.codec.frame.Delimiters;
008    import org.jboss.netty.handler.stream.ChunkedWriteHandler;
009    
010    /**
011     * {@link ChannelPipelineFactory} which is used for the SMTP Client
012     * 
013     * @author Norman Maurer
014     * 
015     *
016     */
017    public class SMTPClientPipelineFactory implements ChannelPipelineFactory{
018    
019        private final static DelimiterBasedFrameDecoder FRAMER = new DelimiterBasedFrameDecoder(8192, true, Delimiters.lineDelimiter());
020        private final static SMTPResponseDecoder SMTP_RESPONSE_DECODER = new SMTPResponseDecoder();
021        private final static SMTPRequestEncoder SMTP_REQUEST_ENCODER = new SMTPRequestEncoder();
022        private final static SMTPClientHandler SMTP_CLIENT_HANDLER = new SMTPClientHandler();
023        private final static SMTPClientIdleHandler SMTP_CLIENT_IDLE_HANDLER = new SMTPClientIdleHandler();
024        @Override
025        public ChannelPipeline getPipeline() throws Exception {
026            ChannelPipeline pipeline = Channels.pipeline();
027            pipeline.addLast("smtpIdleHandler", SMTP_CLIENT_IDLE_HANDLER);
028            pipeline.addLast("framer", FRAMER);
029            pipeline.addLast("decoder", SMTP_RESPONSE_DECODER);
030            pipeline.addLast("encoder", SMTP_REQUEST_ENCODER);
031            pipeline.addLast("chunk", new ChunkedWriteHandler());
032            pipeline.addLast("coreHandler", SMTP_CLIENT_HANDLER);
033            return pipeline;
034        }
035    
036    }
