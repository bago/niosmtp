002    package me.normanmaurer.niosmtp.impl.internal;
003    
004    import java.io.FilterInputStream;
005    import java.io.IOException;
006    import java.io.InputStream;
007    
008    
009    public class DataTerminatingInputStream extends FilterInputStream {
010    
011        private int last;
012        private byte[] extraData;
013        private int pos = 0;
014        private boolean complete = false;
015    
016        private boolean endOfStream = false;
017    
018        public DataTerminatingInputStream(InputStream in) {
019            super(in);
020        }
021    
022        @Override
023        public int read(byte[] b, int off, int len) throws IOException {
024            if (endOfStream == false) {
025    
026                int r = in.read(b, off, len);
027                if (r == -1) {
028                    endOfStream = true;
029                    calculateExtraData();
030    
031                    return fillArray(b, off, len);
032                } else {
033                    // Make sure we respect the offset. Otherwise it could let the RETRCmdHandler
034                    // hang forever. See JAMES-1222
035                    last = b[off + r - 1];
036                    return r;
037                }
038            } else {
039                return fillArray(b, off, len);
040            }
041        }
042    
043        private int fillArray(byte[] b, int off, int len) {
044            int a = -1;
045            int i = 0;
046            if (complete) {
047                return -1;
048            }
049            while (i < len) {
050                a = readNext();
051                if (a == -1) {
052                    complete = true;
053                    break;
054                } else {
055                    b[off + i++] = (byte) a;
056    
057                }
058            }
059            return i;
060        }
061    
062        @Override
063        public int read(byte[] b) throws IOException {
064            return read(b, 0, b.length);
065        }
066    
067        @Override
068        public int read() throws IOException {
069            if (endOfStream == false) {
070                int i = super.read();
071                if (i == -1) {
072                    endOfStream = true;
073                    calculateExtraData();
074                    return readNext();
075                } else {
076                    last = i;
077                }
078                return i;
079    
080            } else {
081                return readNext();
082            }
083        }
084    
085        private void calculateExtraData() {
086            if (last == '\n') {
087                extraData = new byte[3];
088                extraData[1] = '.';
089                extraData[2] = '\r';
090                extraData[3] = '\n';        
091            } else if (last == '\r') {
092                extraData = new byte[4];
093                extraData[0] = '\n';
094                extraData[1] = '.';
095                extraData[2] = '\r';
096                extraData[3] = '\n';
097    
098            } else {
099                extraData = new byte[5];
100                extraData[0] = '\r';
101                extraData[1] = '\n';
102                extraData[2] = '.';
103                extraData[3] = '\r';
104                extraData[4] = '\n';
105            }
106    
107        }
108    
109        private int readNext() {
110            if (extraData == null || extraData.length == pos) {
111                return -1;
112            } else {
113                return extraData[pos++];
114            }
115        }
116    }
