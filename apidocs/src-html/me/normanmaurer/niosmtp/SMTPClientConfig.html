001    package me.normanmaurer.niosmtp;
002    
003    import java.net.InetAddress;
004    import java.net.InetSocketAddress;
005    
006    /**
007     * Configuration which is used to deliver email via SMTP
008     *
009     * @author Norman Maurer
010     *
011     */
012    public interface SMTPClientConfig {
013    
014        
015        /**
016         * Return the name which will get used for the HELO/EHLO
017         * 
018         * @return name
019         */
020        public String getHeloName();
021        
022        /**
023         * Return the connection timeout (in seconds) for the client
024         * 
025         * @return connectionTimeout
026         */
027        public int getConnectionTimeout();
028        
029        /**
030         * Return the response timeout (in seconds) for the SMTP Server to send the response
031         * 
032         * @return responseTimeout
033         */
034        public int getResponseTimeout();
035    
036        
037        /**
038         * Return the {@link InetAddress} which should get used to bind to or null if no specific should get used
039         * 
040         * @return local
041         */
042        public InetSocketAddress getLocalAddress();
043     
044        /**
045         * Return <code>true</code> if the client should use PIPELINING if possible
046         * 
047         * @return pipelining
048         */
049        public boolean usePipelining();
050    }
